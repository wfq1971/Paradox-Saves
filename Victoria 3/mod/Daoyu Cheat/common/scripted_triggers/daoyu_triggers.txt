daoyu_single_culture_trigger = {
	or = {
		and = {
			has_global_variable = daoyu_add_homeland_for_mark_target_culture_trigger
			has_global_variable = dculo_$CULTURE$
			any_state = {
				has_variable = daoyu_target_state
				state_region = { not = { is_homeland = cu:$CULTURE$ } }
			}
		}
		and = {
			has_global_variable = daoyu_homeland_of_non_primary_culture_trigger
			any_scope_state = {
				has_decree = decree_daoyu_target_state
				root = { not = { country_has_primary_culture = cu:$CULTURE$ } }
				state_region = { is_homeland = cu:$CULTURE$ }
			}
		}
	}
}
daoyu_cultural_obsession_trigger = {
	or = {
		and = {
			has_global_variable = daoyu_add_cultural_obsession
			any_country = { leading_producer_of = g:$GOODS$ }
			scope:daoyu_target_country = { any_scope_culture = { is_primary_culture_of = scope:daoyu_target_country not = { has_cultural_obsession = $GOODS$ } } }
		}
