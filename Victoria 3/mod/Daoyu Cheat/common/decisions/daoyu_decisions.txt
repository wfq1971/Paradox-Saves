daoyu_territorial_integrity = {
	is_shown = {
		is_ai = no
		not = { has_global_variable = daoyu_off }
		game_date = 1836.1.1
		any_state_region = {
			or = {
				and = { any_scope_state = { owner = root is_incorporated = yes is_treaty_port = no } not = { any_scope_state = { owner = { is_country_type = decentralized } } } }
				any_scope_state = { owner = root is_homeland_of_country_cultures = root }
			}
			any_scope_state = { not = { owner = root } }
		}
	}
	possible = {
		any_state_region = {
			save_temporary_scope_as = imperfect_region
			or = {
				and = { any_scope_state = { owner = root is_incorporated = yes is_treaty_port = no } not = { any_scope_state = { owner = { is_country_type = decentralized } } } }
				any_scope_state = { owner = root is_homeland_of_country_cultures = root }
			}
			any_scope_state = { not = { owner = root } }
		}
		game_date = 1836.1.1
	}
	when_taken = {
		every_state_region = {
			limit = {
				or = {
					and = { any_scope_state = { owner = root is_capital = no is_treaty_port = no } not = { any_scope_state = { owner = { is_country_type = decentralized } } } }
					any_scope_state = { owner = root is_capital = yes }
				}
				any_scope_state = { not = { owner = root } }
			}
			every_scope_state = {
				limit = { not = { owner = root } }
				set_state_owner = root
			}
		}
	}
}
